Project overview

  • Chatbot: Uses microsoft/DialoGPT-small for conversational replies.
  • Model download: Fetches pre‚Äëtrained models from Hugging Face and saves locally.
  • Command routing: Distinguishes between conversation and download <model-name> commands.

    pip install transformers requests
  • 2. Save script: Copy the code into chatbot.py.
  • 3. Run:
    python chatbot.py
  • 4. Use: Type normal text to chat, or:
    download bert-base-uncased
chatbot.py
from transformers import pipeline
import requests
import os

# Create a chatbot using a small pretrained model
chatbot = pipeline("conversational", model="microsoft/DialoGPT-small")

def chat_response(user_input):
    conversation = chatbot(user_input)
    return conversation[0]["generated_text"]

def download_model(model_name, save_dir="models"):
    os.makedirs(save_dir, exist_ok=True)
    url = f"https://huggingface.co/{model_name}/resolve/main/pytorch_model.bin"
    response = requests.get(url)
    if response.status_code == 200:
        file_path = os.path.join(save_dir, f"{model_name.replace("/", "_")}.bin")
        with open(file_path, "wb") as f:
            f.write(response.content)
        return f"‚úÖ Model {model_name} downloaded successfully to {file_path}"
    else:
        return f"‚ùå Failed to download {model_name}. Status: {response.status_code}"

def chatbot_interface():
    print("AI Model Chatbot ü§ñ (type 'exit' to quit)")
    while True:
        user_input = input("You: ")
        if user_input.lower() == "exit":
            break
        elif user_input.startswith("download"):
            # Example: "download bert-base-uncased"
            try:
                _, model_name = user_input.split(" ", 1)
                print(download_model(model_name.strip()))
            except ValueError:
                print("Please use: download <model-name>")
        else:
            print("Bot:", chat_response(user_input))

if __name__ == "__main__":
    chatbot_interface()
Note: This is a CLI demo. It requires internet access to download models and may take time depending on size.
